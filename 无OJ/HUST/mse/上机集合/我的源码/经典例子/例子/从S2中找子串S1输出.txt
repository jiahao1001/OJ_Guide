      int search(char s1[],char s2[])
	  {
		  int i=0,j,len=strlen(s2);
	  while(s1[i])
	  {
		  for(j=0;j<len;j++)
			  if(s1[i+j]!=s2[j]) break;
			  if(j>=len)return i;
			  else i++;
	  }
	  return -1;
}
